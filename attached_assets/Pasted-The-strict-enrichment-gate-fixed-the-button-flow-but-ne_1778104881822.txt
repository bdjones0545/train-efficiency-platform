2. Add shared normalization before saving any lead.

function normalizeNullable(value?: string | null): string | null {
  if (!value) return null;

  const cleaned = value.trim();
  const lowered = cleaned.toLowerCase();

  if (
    lowered === "unknown" ||
    lowered === "n/a" ||
    lowered === "none" ||
    lowered === "not found" ||
    lowered === "unavailable" ||
    lowered === "null" ||
    lowered === "-"
  ) {
    return null;
  }

  return cleaned;
}

5. Add domain extraction helper.

function extractDomainFromUrl(url?: string | null): string | null {
  if (!url) return null;

  try {
    const parsed = new URL(url.startsWith("http") ? url : `https://${url}`);
    return parsed.hostname.replace(/^www\./, "");
  } catch {
    return null;
  }
}

function inferEmailFromDomain(domain: string, leadType?: string | null, sport?: string | null): string | null {
  const normalizedSport = sport?.toLowerCase() ?? "";
  const normalizedType = leadType?.toLowerCase() ?? "";

  let localPart = "info";

  if (normalizedType.includes("academy") || normalizedType.includes("school")) {
    localPart = "athletics";
  } else if (normalizedSport.includes("basketball")) {
    localPart = "coach";
  } else if (normalizedSport.includes("baseball") || normalizedSport.includes("soccer") || normalizedSport.includes("football")) {
    localPart = "coach";
  }

  return `${localPart}@${domain}`;
}

function isValidEmail(value?: string | null): boolean {
  if (!value) return false;
  return /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(value.trim());
